def normalize(s)
    s.downcase.gsub(%r[ä]/, "ae").gsub(%r[ö]/, "oe").gsub(%r[ü]/, "ue").gsub(%r[ß]/, "ss").gsub(%r[^a-zA-Z\ ]/,"").split
end
